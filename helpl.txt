func ConvertToUserMinimal(users []*User) []*UserMinimal {
    result := make([]*UserMinimal, 0, len(users))
    for _, user := range users {
        result = append(result, &UserMinimal{
            FirstName: user.FirstName,
            LastName:  user.LastName,
        })
    }
    return result
}
